# :hammer: HC165 扫描按键

## 关联库

```c
#include "hc165.h"
```

## 配置

```c
#define DATA 25 //数据
#define CLOCK 26 //时钟
#define LATCH 27 //latch引脚
#define key_nums 8 //按键数量

//hc165结构体
HC165_typeDef hc165 = {
  .data_pin = DATA,
  .clock_pin = CLOCK,
  .latch_pin = LATCH,
  .key_num = key_nums,
};

//按键显示内容(非必要)
KEYMAP_typeDef key_map[] = 
{
  {'1'},
  {'2'},
  {'3'},
  {'4'},
  {'5'},
  {'6'},
  {'7'},
  {'8'},
};

//实体化按键
BTN165_typeDef btn165[key_nums];


void  press_handler()
{
  //按下函数(一般用于键盘)  
  Serial.println("press:(in main.cpp)");
}

void  release_handler()
{
  //释放函数  
  Serial.println("release:(in main.cpp)");
};

```

## 使用
```arduino
void setup() {
  hc165_init(&hc165,btn165,key_nums);
  
}

void loop() {
  hc165_scan(&hc165,key_map,btn165);
}
```
